# 数据验证
数据验证的位置
* 前端（客户端）：为了用户体验
* 路由层：验证接口格式是否正常
* 业务逻辑层：保证业务完整性
* 数据库验证（约束）：保证数据完整性

相关的库：

[validator](https://github.com/validatorjs/validator.js) 用于验证某个字符串是否满足某个规则

[validate.js](http://validatejs.org/) 用于验证某个对象的树形是否满足某些规则

其中，对于datetime，必须扩展其parse和format方法

```js
/* service层初始化服务 */
const moment = require("moment");
const validate = require("validate.js");

validate.extend(validate.validators.datetime, {
    /**
     * 该函数会自动用于日期格式转换
     * 它会在验证时自动触发，它需要将任何数据转换为时间戳返回
     * 如果无法转换，返回NaN
     * @param {*} value 传入要转换的值
     * @param {*} options 针对某个属性的验证配置
     */
    parse(value, options) {
        let formats = ['YYYY-MM-DD HH:mm:ss', 'YYYY-M-D H:m:s', 'x'];

        // validate.js 传进来的就是 datetime 规则本身，不会再包一层 datetime
        if (options?.dateOnly) {
            formats = ['YYYY-MM-DD', 'YYYY-M-D', 'x'];
        }

        return +moment.utc(value, formats, true);
    },

    /**
     * 用户显示错误消息时，使用的显示字符串
     */
    format(value, options) {
        let format = "YYYY-MM-DD";

        if (!options.dateOnly) {
            format += " HH:mm:ss";
        }

        return moment.utc(value).format(format);
    },
})
```

## 最佳实践
现在可知道，我们会对项目分为三层：model service route
那么，每层往往都会创建一个init文件，来做初始化相关的逻辑。
```sh
├─ models
│  ├─ db.js    数据库连接
│  ├─ sync.js  数据库同步
│  ├─ init.js  数据库初始化
│  └─ user.js  用户模型
├─ services
│  ├─ userServices.js  用户服务层
│  └─ init.js  服务初始化
├─ routes
│  ├─ init.js  路由初始化
│  └─ loginRoutes.js  登录路由
├─  index.js
└─  init.js
```

```js  [init.js]
// 这里对应三个层面的init.js

// model层
require("./relation");
require("./sync");

// service层
//xxx

// init层
require("./models/init");
require("./services/init");
```

```js [utils/helper.js]
/* 从对象中提取指定的属性值 */
exports.pick = function (obj, ...props) {
    if (!obj || typeof obj !== "object") {
        return obj;
    }
    const newObj = {};
    for (const key in obj) {
        if (props.includes(key)) {
            newObj[key] = obj[key];
        }
    }
    return newObj;
};
```

```js [services/student.js]
const Student = require("../models/Student.js");
const Class = require("../models/Class");
const { Op } = require("sequelize");

const validate = require("validate.js");
const moment = require("moment");
const { pick } = require("../utils/helper.js");

/* 添加学生 */
exports.addStudent = async (studentObj) => {
    /* 提取需要的属性，防止添加可以添加deletedAt这种字段 */
    const newStu = pick(studentObj, "name", "sex", "birthdady", "mobile")

    /* 自定义校验规则 */
    validate.validators.classExits = async (value) => {
        const c = await Class.findByPk(value);
        if (c) {
            return;
        }
        return "is not exist";
    }

    /* 定义校验规则 */
    const rules = {
        name: {
            presence: { allowEmpty: false },
            length: {
                minimum: 2,
                maximum: 10,
            }
        },
        birthdady: {
            presence: { allowEmpty: false },
            datetime: {
                dateOnly: true,
                earliest: +moment.utc().subtract(100, "years"),
                latest: +moment.utc().subtract(5, "years")
            }
        },
        sex: {
            presence: { allowEmpty: false },
            type: "boolean",
        },
        mobile: {
            presence: { allowEmpty: false },
            type: "string",
            format: /1\d{10}/
        },
       ClassId: {
            presence: { allowEmpty: false },
            numericality: {
                onlyInteger: true,
                greaterThan: 0
            },
            classExits: true
        }
    }

    /* 校验 */
    // const error = validate.validate(studentObj, rules);
    // console.log("错误了==>", error);
    await validate.async(newStu, rules);//异步校验

    const ins = await Student.create(newStu)
    return ins.toJSON()
}

exports.delStudent = async (id) => {
    const res = await Student.destroy({
        where: {
            id,
        }
    })
    return res
}

exports.updateStudent = async (id, studentObj) => {
    const res = await Student.update(studentObj, {
        where: {
            id,
        }
    })
    return res
}

exports.getStudentById = async (id) => {
    const stu = await Student.findByPk(id);
    if (!stu) { return null };
    return stu.toJSON();
}

exports.getAllStudents = async () => {
    const students = await Student.findAll();
    if (!students) { return null };
    return students.map(item => item.toJSON());
}

/* 分页查询 */
exports.getStudents = async (page = 1, size = 10, sex = -1, name = "") => {
    // const result = await Student.findAll({

    //     offset: (page - 1) * size
    // })
    // if (!result) { return null };
    // const total = await Student.count();

    const where = {};
```
