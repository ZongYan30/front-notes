# 模型定义和同步

## 服务三层架构

models:数据库模型
services:服务层,负责处理业务
routes:路由层,负责定义接口侧

```sh
├─ models
│  ├─ db.js    数据库连接
│  ├─ sync.js  数据库同步
│  └─ user.js  用户模型
├─ services
│  └─ userServices.js  用户服务层
└─ routes
   └─ loginRoutes.js  登录路由
```

```js [adminService.js]
const Admin = require("../models/Admin.js");

/* 新增 */
exports.addAdmin = async (adminObj) => {
  // 此处做各种权限判断和其它业务逻辑
  // 应该判断adminObj的各种属性是否合理，以及账号是否已存在

  /* 法一:手动保存 */
  const ins = Admin.build({
    loginId: adminObj.loginId,
    loginPwd: adminObj.loginPwd,
  });

  ins.loginId = "bcd";

  ins.save().then(() => {
    console.log("新建管理员成功");
  });

  /* 法二:自动保存 */
  const ins = await Admin.create(adminObj); //创建得到实例
  return ins.toJSON();
};

/* 删除 */
exports.delAdmin = async (adminId) => {
  /* 法一:找到实例 */
  /* 	const ins = await Admin.findByPk(adminId); //通过主键找到对应数据实例

		console.log(ins);

		if (ins) {
			await ins.destroy();
		} 
	*/

  /* 法二：通过表实例删除 */
  const res = await Admin.destroy({
    where: {
      id: adminId,
    },
  });
  return res;
};

/* 修改 */
exports.updateAdmin = async (id, adminObj) => {
  /* 法一：得到实例 */
  // const ins = await Admin.findByPk(id);
  // ins.loginId = adminObj.loginId;
  // ins.save(); //保存这次操作

  /* 法二：表实例操作 */
  const res = await Admin.update(adminObj, {
    where: {
      id,
    },
  });

  return res;
};
```

