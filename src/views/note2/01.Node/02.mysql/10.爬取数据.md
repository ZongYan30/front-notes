# 爬取数据

爬取数据需要借助两个第三方库 [axios](https://github.com/axios/axios) 和 [cheeriojs](https://github.com/cheeriojs/cheerio#readme)

> cheeriojs是一个类似JQuery的库,方便抓取页面信息。

## 安装依赖

```sh
npm install axios cheeriojs -S
```

## 使用

新建spider.js

```js
/* 豆瓣书籍爬虫 */
const axios = require("axios").default;
const cheerio = require("cheerio");
const Book = require("../models/Book.js");

/**
 * 获取豆瓣html结构
 */
const getDoubanHtml = async (url) => {
  const res = await axios.get(url, { timeout: 120000 });
  return res.data;
};

/**
 * 解析豆瓣html结构
 */
const parseHtml = async () => {
  const pageHtml = await getDoubanHtml("https://book.douban.com/latest");
  const $ = cheerio.load(pageHtml); //将页面转为JQuery对象
  const achorElements = $(".chart-dashed-list .media__img a");

  const linkUrls = achorElements
    .map((_index, ele) => {
      const href = ele.attribs["href"];
      return href;
    })
    .get();

  return linkUrls;
};

/**
 * 获取图书详情
 */
const getBookDetail = async (url) => {
  const res = await axios.get(url, { timeout: 120000 });
  const $ = cheerio.load(res.data);

  const bookName = $("h1").text().trim();
  const imgurl = $("#mainpic .nbg img").attr("src");
  const spans = $("#info span.pl");
  const authorSpan = spans.filter((i, ele) => {
    return $(ele).text().includes("作者");
  });
  const author = authorSpan.next("a").text();
  const publishSpan = spans.filter((i, ele) => {
    return $(ele).text().includes("出版年");
  });
  const publishDate = publishSpan[0].nextSibling.nodeValue.trim();

  return {
    name: bookName,
    imgurl,
    publishDate,
    author,
  };
};

/**
 * 获取全部图书数据
 */
const fetchAll = async () => {
  const links = await parseHtml();
  const proms = links.map((link) => getBookDetail(link));
  return Promise.all(proms);
};

/**
 * 保存数据到数据库
 */

const saveToDB = async (listArr) => {
  await Book.bulkCreate(listArr);
  console.log("数据爬取成功且保存到数据库");
};

/**
 * 入口函数
 */
const main = async () => {
  const listArr = await fetchAll();
  saveToDB(listArr);
};

main();
```
