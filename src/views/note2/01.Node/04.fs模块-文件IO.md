# Node.js fs 模块 — 文件 IO

---

## 一、fs.promises API

Node.js 的 `fs` 模块提供了 `fs.promises` 命名空间，所有 API 均返回 Promise，可使用 `async/await` 调用。

### 目录操作

| API | 说明 | 参数 |
|---|---|---|
| `fs.promises.mkdir(path)` | 创建目录 | `path`：目录路径 |
| `fs.promises.readdir(path)` | 读取目录内容，返回文件名数组 | `path`：目录路径 |

**示例**

```js
const fs = require("fs");
const path = require("path");

const filePath = path.join(__dirname, "./myfiles");

// 创建目录
(async () => {
  await fs.promises.mkdir(filePath);
  console.log("创建目录成功");
})();

// 读取目录
(async () => {
  const content = await fs.promises.readdir(filePath);
  console.log("读取目录成功", content); // ["1.txt", "a", ...]
})();
```

---

### 文件读写

| API | 说明 | 参数 |
|---|---|---|
| `fs.promises.readFile(path, encoding?)` | 读取文件内容 | `path`：文件路径；`encoding`：字符编码（如 `"utf-8"`），不传则返回 Buffer |
| `fs.promises.writeFile(path, data)` | 写入文件（覆盖写入） | `path`：文件路径；`data`：字符串或 Buffer |

**示例**

```js
const target = path.join(filePath, "1.txt");

// 读取文件（utf-8 编码，直接返回字符串）
const content = await fs.promises.readFile(target, "utf-8");
console.log("读取文件成功", content);

// 写入字符串
await fs.promises.writeFile(target, "Hello world");

// 写入 Buffer
const buffer = Buffer.from("second write", "utf-8");
await fs.promises.writeFile(target, buffer);

console.log("写入文件成功");
```

---

### 获取文件信息

| API | 说明 | 返回值常用属性/方法 |
|---|---|---|
| `fs.promises.stat(path)` | 获取文件/目录的状态信息 | `isFile()`、`isDirectory()`、`size`、`birthtime`、`mtime` |

**示例**

```js
const info = await fs.promises.stat(target);

console.log(info.isFile());       // true
console.log(info.isDirectory());  // false
console.log(info.size);           // 文件大小（字节）
```

---

### 复制文件（综合示例）

结合 `stat`、`readFile`、`writeFile` 实现文件复制：

```js
const copyFile = async (filePath, fileName) => {
  const stat = await fs.promises.stat(filePath);

  if (stat.isFile()) {
    const content = await fs.promises.readFile(filePath, "utf-8");
    const buffer = Buffer.from(content, "utf-8");
    const newPath = path.join(filePath, "../", fileName);
    await fs.promises.writeFile(newPath, buffer);
    console.log("复制文件成功");
  }
};

const file = path.join(filePath, "1.txt");
copyFile(file, "1.copy.txt");
```

## 二、综合实践 — File 类

基于 `fs.promises` API 封装一个 `File` 类，实现面向对象的文件操作。

### 类定义

```js
class File {
  constructor(filename, name, ext, isFile, size, createTime, updateTime) {
    this.filename = filename;
    this.name = name;
    this.ext = ext;
    this.isFile = isFile;
    this.size = size;
    this.createTime = createTime;
    this.updateTime = updateTime;
  }

  // 获取文件内容（isBuffer 为 true 时返回 Buffer，否则返回字符串）
  async getContent(isBuffer) {
    return await fs.promises.readFile(this.filename, !isBuffer && "utf-8");
  }

  // 静态方法：根据路径创建 File 实例
  static async getFile(filename) {
    const stat = await fs.promises.stat(filename);
    const name = path.basename(filename);
    const ext = path.extname(filename);
    const isFile = stat.isFile();
    const size = stat.size;
    const createTime = new Date(stat.birthtime).toLocaleString();
    const updateTime = new Date(stat.mtime).toLocaleString();
    return new File(filename, name, ext, isFile, size, createTime, updateTime);
  }

  // 获取子文件列表（仅目录有效）
  async getChildren() {
    if (this.isFile) {
      return [];
    }
    const nameArr = await fs.promises.readdir(this.filename);

    const children = nameArr.map(name => {
      const namePath = path.resolve(this.filename, name);
      return File.getFile(namePath);
    });

    return Promise.all(children);
  }
}
```

### 使用示例

```js
// 工具函数：读取目录下的所有文件信息
async function readDir(dirname) {
  const dir = await File.getFile(dirname);
  return await dir.getChildren();
}

// 使用 File 类读取 myfiles/a/a.txt 的内容
(async () => {
  const filePath = path.join(__dirname, "./myfiles");
  const res = await readDir(filePath);
  const datas = await res[res.length - 1].getChildren();
  const content = await datas[0].getContent();
  console.log("文件内容==>", content);
})();
```
