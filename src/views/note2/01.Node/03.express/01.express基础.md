# express
node中的框架有express,koa2等。
经典的就是[express](http://expressjs.com/)  [中文网](https://www.expressjs.com.cn/)

## 使用
```sh
npm i express -S
```

```js
const express = require("express");
const app = express(); //创建一个express应用

app.listen(9527, () => {
  console.log(`server listen on 9527`);
});
```

另一种方法

```js
const express = require("express");
const http = require("http");

const app = express(); //创建一个express应用


const server = http.createServer(app);
const port = 5008;

server.listen(port, () => {
  console.log(`server listen on ${port}`);
});
```

## 请求和响应
```js
app.get("/news/:id", (req, res) => {
  // req 和 res 是被express封装过后的对象
  
  // 请求
  console.log("请求头", req.headers); //获取请求头，对象
  console.log("请求路径", req.path);
  console.log("query", req.query);
  console.log("params", req.params);

  // 响应
  res.setHeader("a", "123");//设置响应头

  res.send([2, 3, 4]);//自动设置响应mime类型
  res.send({ id: 123,name: "成哥",age: 18});

  res.status(302).header("location", "https://duyi.ke.qq.com").end();
  res.status(302).location("https://duyi.ke.qq.com").end();
  res.redirect(302, "https://duyi.ke.qq.com");//重定向
});
```

```js
// 匹配任何get请求
app.get("*", (req, res) => {
  console.log("abc");
});
```
